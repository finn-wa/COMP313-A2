tree "Root"
	while not Paused()
		fallback
			sequence
				InSight()
				tree "Attack"
			sequence
				HasArrived()
				tree "Patrol"

tree "Patrol"
	MoveToNextPoint()

tree "Attack"
	repeat
		fallback
			fallback
				sequence
					InRange()
					Attack()
				sequence
					InSight()
					Chase()
			tree "Search"

tree "Search"
	sequence
		SetTimer(7.0)
		while not TimerFinished()
			fallback
				sequence 
					InSight()
					SetTimer(-1.0)
				not
					mute
						fallback
							sequence
								TimeRemaining(3.5)
								Chase()
							sequence
								Follow()
